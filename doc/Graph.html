def add_edge(node_from_key, node_to_key, value)
  edge_key = [node_from_key, node_to_key]
  if self.exist_nodes(edge_key)
    @edges[edge_key] = value
  else
    return false
  end
end

def exist_nodes(key_arr)
  key_arr.each do |key|
    if not @nodes.has_key?(key)
      return false
    end
  end
  return true
end
